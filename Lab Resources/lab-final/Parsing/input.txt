#include <stdio.h>
int main() {
 char op;
 int a = 10, b = 5;
 printf("Enter an operator (+, -, *, /): ");
 scanf(" %c", &op);
 switch (op) {
 case '+':
 printf("Result: %d\n", a + b);
 break;
 case '-':
 printf("Result: %d\n", a - b);
 break;
 case '*':
 printf("Result: %d\n", a * b);
 break;
 case '/':
 if (b != 0)
 printf("Result: %d\n", a / b);
 else
 printf("Error: Division by zero\n");
 break;
 default:
 printf("Unknown operator\n");
 }
 return 0;
}

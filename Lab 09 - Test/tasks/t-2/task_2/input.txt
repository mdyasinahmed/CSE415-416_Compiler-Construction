int main()
{
    char op;
    double a, b, result;
    printf("Enter operation (+, -, *, /, ^): ");
    scanf(" %c", &op);
    printf("Enter two numbers: ");
    scanf("%lf %lf", &a, &b);
    switch (op)
    {
    case '+':
        result = a + b;
        printf("Result: %.2lf\n", result);
        break;
    case '-':
        result = a - b;
        printf("Result: %.2lf\n", result);
        break;
    case '*':
        result = a * b;
        printf("Result: %.2lf\n", result);
        break;
    case '/':
        if (b != 0)
            printf("Result: %.2lf\n", a / b);
        else
            printf("Error: Division by zero!\n");
        break;
    case '^':
        result = pow(a, b);
        printf("Result: %.2lf\n", result);
        break;
    default:
        printf("Invalid operator!\n");
    }
    return 0;
}
